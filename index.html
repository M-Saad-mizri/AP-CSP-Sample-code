// This code was written and modified by a student
// for AP CSP Create Performance Task demonstration.

// List (Collection)
const scores = [56, 72, 88, 45, 91, 67, 100];

// Student-developed procedure
function getHighScores(threshold) {
  const result = [];
  for (let score of scores) {
    if (score > threshold) {
      result.push(score);
    }
  }
  return result;
}

// Input + Output
function showHighScores() {
  const userValue = Number(document.getElementById("userInput").value);
  const high = getHighScores(userValue);
  const output = document.getElementById("output");

  if (high.length > 0) {
    output.textContent = `Scores above ${userValue} are: ${high.join(", ")}`;
  } else {
    output.textContent = `No scores above ${userValue}.`;
  }
}

// Attach event listener
document.getElementById("runButton").addEventListener("click", showHighScores);
